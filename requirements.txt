# requirements.txt — List of Python packages this project needs
#
# HOW TO INSTALL:
#   Run this command in your terminal (inside the daily-curator folder):
#   pip install -r requirements.txt
#
# WHAT THIS MEANS:
#   "pip" is Python's package manager — like an app store for Python libraries.
#   The -r flag means "install everything listed in this requirements file".
#   Each line below is one package to install.
#
# ─────────────────────────────────────────────────────────────────────────────

# anthropic — The official Python library for talking to Claude AI
# This lets us send requests to Claude and receive responses
anthropic>=0.40.0

# requests — A simple library for making web requests (like visiting a URL)
# We use this to call the Inoreader API
requests>=2.31.0

# python-dotenv — Reads .env files and loads them into Python's environment
# This lets us store API keys in a .env file instead of hardcoding them
python-dotenv>=1.0.0

# pywebpush — Sends Web Push notifications to subscribed browsers using VAPID keys
# Used by send_push.py to deliver briefing notifications after each curator run
pywebpush>=2.0.0

# feedparser — Parses RSS and Atom feeds (used by Direct RSS fetch alongside Inoreader)
feedparser>=6.0.0

# PyNaCl — libsodium bindings used by get_inoreader_token.py to encrypt secrets
# before sending them to the GitHub Actions secrets API (required by GitHub)
PyNaCl>=1.5.0

# Pillow — image processing library used by image_sourcer.py
# Handles open/crop/resize/draw for Instagram-format output images
Pillow>=10.0.0

# beautifulsoup4 — HTML parser used by image_sourcer.py
# Scrapes og:image meta tags from article pages
beautifulsoup4>=4.12.0
